poniższy kod pozwala na wylosowanie dowolnej frazy, z bazy wprowadzonej do programu! został napisany w języku python. dodatkowo można wybrać ile wyrazów zostanie uwzględnione w wyniku. skrypt można zobaczyć/pobrać na github'ie pod tym linkiem.

from random import randint

def losuj():
    score = randint(0, len(phrase))
    end = phrase[score - 1]
    str(end)
    print('Wylosowana fraza to: ' + end)
                        
inp = str(input('Podaj frazy [oddziel srednikami, bez spacji]: '))
inp2 = int(input('Ile fraz chcesz wylosowac?: '))
                        
phrase = inp.split(";")
                        
for i in range(0, inp2):
    losuj()
